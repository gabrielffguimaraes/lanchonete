create table client(
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) UNIQUE NOT NULL,
    password TEXT NOT NULL,
    complete_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    role VARCHAR(20),
    created_at DATETIME DEFAULT NOW()
);

create table ingredient(
    id INT PRIMARY KEY AUTO_INCREMENT,
    description VARCHAR(255) UNIQUE NOT NULL
);

create table product(
    id INT PRIMARY KEY AUTO_INCREMENT,
    description VARCHAR(255),
    category INT,
    price DECIMAL(10,2),
    price_fake DECIMAL(10,2),
    detail TEXT,
    review TEXT,
    sit BOOLEAN DEFAULT true
);

create table product_ingredient(
    id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT NOT NULL,
    ingredient_id INT NOT NULL
);


create table status(
    id INT PRIMARY KEY,
    description VARCHAR(255) UNIQUE NOT NULL,
    last BOOLEAN

create table product_photo(
    id INT PRIMARY KEY AUTO_INCREMENT,
    product_id INT NOT NULL,
    name VARCHAR(255) NOT NULL,
    type VARCHAR(255) NOT NULL
);

create table payment_order (
   id INT PRIMARY KEY AUTO_INCREMENT,
   client_id INT NOT NULL,
   address_id INT NOT NULL,
   discount DECIMAL(10,2),
   delivery_fee DECIMAL(10,2),
   status INT NOT NULL,
   created_at datetime default now()
);

create table payment_order_product (
    id INT PRIMARY KEY AUTO_INCREMENT,
    payment_order_id INT NOT NULL,
    product_id INT NOT NULL,
    quantity INT NOT NULL,
    price DECIMAL(10,2)
);
create table payment_order_product_ingredient (
    id INT PRIMARY KEY AUTO_INCREMENT,
    payment_order_product_id INT NOT NULL,
    ingredient_id INT NOT NULL
);
create table payment_order_status (
    id INT PRIMARY KEY AUTO_INCREMENT,
    payment_order_id INT NOT NULL,
    status INT NOT NULL,
    created_at datetime default now()
);

create table address (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name varchar(255),
    cep varchar(255),
    address TEXT,
    complement TEXT,
    city TEXT,
    uf VARCHAR(2),
    country varchar(255),
    client_id INT NOT NULL,
    ativo BOOLEAN default true
);
